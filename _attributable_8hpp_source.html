164 inline bool
165 Attributable::setAttribute(std::string const& key, T&& value)
166 {
167  dirty = true;
168  auto it = m_attributes->lower_bound(key);
169  if( it != m_attributes->end() && !m_attributes->key_comp()(key, it->first) )
170  {
171  // key already exists in map, just replace the value
172  it->second = Attribute(value);
173  return true;
174  } else
175  {
176  // emplace a new map element for an unknown key
177  m_attributes->emplace_hint(it,
178  std::make_pair(key, Attribute(value)));
179  return false;
180  }
181 }

195 template< typename T >
196 inline T
197 Attributable::readFloatingpoint(std::string const& key) const
198 {
199  static_assert(std::is_floating_point< T >::value, "Type of attribute must be floating point");
200 
201  T t{0};
202  Attribute a = getAttribute(key);
203  Datatype target_dtype = determineDatatype< T >();
204  if( a.dtype == target_dtype )
205  t = a.get< T >();
206  else
207  {
208  warnWrongDtype(key, a.dtype, target_dtype);
209  using DT = Datatype;
210  switch( a.dtype )
211  {
212  case DT::FLOAT:
213  t = static_cast< T >(a.get< float >());
214  break;
215  case DT::DOUBLE:
216  t = static_cast< T >(a.get< double >());
217  break;
218  case DT::LONG_DOUBLE:
219  t = static_cast< T >(a.get< long double >());
220  break;
221  default:
222  throw std::runtime_error("Unknown floating point datatype.");
223  }
224  }
225  return t;
226 }

242 Attributable::readVectorFloatingpoint(std::string const& key) const

244  static_assert(std::is_floating_point< T >::value, "Type of attribute must be floating point");
245 
246  std::vector< T > vt{};

251  else
252  {
253  warnWrongDtype(key, a.dtype, target_dtype);
254  using DT = Datatype;
255  switch( a.dtype )
256  {
257  case DT::VEC_FLOAT:
258  for( auto const& val : a.get< std::vector< float > >() )
259  vt.push_back(static_cast< T >(val));
260  break;
261  case DT::VEC_DOUBLE:

263  vt.push_back(static_cast< T >(val));
